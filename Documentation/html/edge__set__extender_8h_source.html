588 Edge edge = Parent::addEdge(from, to);
589 notifier(Edge()).add(edge);

606 arcs.push_back(Parent::direct(edge, false));
607 notifier(Arc()).erase(arcs);
608 notifier(Edge()).erase(edge);
609 Parent::erase(edge);
