132 Mat gaussianEliminate(int& sign)
133 {
134 sign = 1;
135
136 Mat result(*this);
137
138 int rows = height;
139 int cols = width;
140 int Acols = cols - 1;
141 float eps = Math::Constant::epsilon;
142
143 int row = 0; // row tracker
144 int col = 0; // column tracker
145
146 // iterate through the rows
147 while (row < rows)
148 {
149 // find a pivot for the row
150 bool pivot_found = false;
151 while (col < Acols && !pivot_found)
152 {
153 if (result(row, col) != 0) { // pivot not equal to 0
154 pivot_found = true;
155 }
156 else { // check for a possible swap
157 int max_row = row;
158 double max_val = 0;
159 for (int k = row + 1; k < rows; ++k)
160 {
161 double cur_abs = result(k, col) >= 0 ? result(k, col) : -1 * result(k, col);
162 if (cur_abs > max_val)
163 {
164 max_row = k;
165 max_val = cur_abs;
166 }
167 }
168 if (max_row != row) {
169 result.swapRows(max_row, row);
170 pivot_found = true;
171 sign *= -1;
172 }
173 else {
174 col++;
175 }
176 }
177 }
178
179 // perform elimination as normal if pivot was found
180 if (pivot_found)
181 {
182 for (int t = row + 1; t < rows; ++t) {
183 for (int s = col + 1; s < cols; ++s) {
184 result(t, s) = result(t, s) - result(row, s) * (result(t, col) / result(row, col));
185 if (result(t, s) < eps && result(t, s) > -1 * eps)
186 result(t, s) = 0;
187 }
188 result(t, col) = 0;
189 }
190 }
191
192 row++;
193 col++;
194 }
195
196 return result;
197 }
