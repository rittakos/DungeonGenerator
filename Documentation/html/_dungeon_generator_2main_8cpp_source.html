317 bool MakeFeature(Map& map, RngT& rng) const
318 {
319 auto tries = 0;
320 auto maxTries = 1000;
321
322 for (; tries != maxTries; ++tries)
323 {
324 // Pick a random wall or corridor tile.
325 // Make sure it has no adjacent doors (looks weird to have doors next to each other).
326 // Find a direction from which it's reachable.
327 // Attempt to make a feature (room or corridor) starting at this point.
328
329 int x = GetRandomInt(rng, 1, XSize - 2);
330 int y = GetRandomInt(rng, 1, YSize - 2);
331
332 if (map.GetCell(x, y) != Tile::DirtWall && map.GetCell(x, y) != Tile::Corridor)
333 continue;
334
335 if (map.IsAdjacent(x, y, Tile::Door))
336 continue;
337
338 if (map.GetCell(x, y + 1) == Tile::DirtFloor || map.GetCell(x, y + 1) == Tile::Corridor)
339 {
340 if (MakeFeature(map, rng, x, y, 0, -1, Direction::North))
341 return true;
342 }
343 else if (map.GetCell(x - 1, y) == Tile::DirtFloor || map.GetCell(x - 1, y) == Tile::Corridor)
344 {
345 if (MakeFeature(map, rng, x, y, 1, 0, Direction::East))
346 return true;
347 }
348 else if (map.GetCell(x, y - 1) == Tile::DirtFloor || map.GetCell(x, y - 1) == Tile::Corridor)
349 {
350 if (MakeFeature(map, rng, x, y, 0, 1, Direction::South))
351 return true;
352 }
353 else if (map.GetCell(x + 1, y) == Tile::DirtFloor || map.GetCell(x + 1, y) == Tile::Corridor)
354 {
355 if (MakeFeature(map, rng, x, y, -1, 0, Direction::West))
356 return true;
357 }
358 }
359
360 return false;
361 }
